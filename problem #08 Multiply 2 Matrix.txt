#include<iostream>
#include<string>
#include<iomanip>
using namespace std;

int Randomnumber(int from, int to)
{
	int randum = rand() % (to - from + 1) + from;
	return randum;
}
void FillMatrixWithRandomNumbers(int arr[3][3], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			arr[i][j] = Randomnumber(1, 100);
		}
	}
}
void MultiplyMatrix(int arr[3][3], int arr2[3][3],int arrMultiply[3][3], short Rows, short Cols)
{
	for (int i = 0; i < Rows; i++)
	{
		for (int  j = 0; j < Cols; j++)
		{
			arrMultiply[i][j]=arr[i][j] * arr2[i][j];
		}
	}
}
void PrintMatrix(int arr[3][3], int Rows, int Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			cout << setw(3) << arr[i][j] << "      ";
		}
		cout << "\n";
	}

}

int main()
{
	//Seeds the random number genrator in c++, called only once
	srand((unsigned)time(NULL));

	int arr[3][3],arr2[3][3],arrMultiply[3][3];

	FillMatrixWithRandomNumbers(arr, 3, 3);

	cout << "\n Matrix One: \n";
	PrintMatrix(arr, 3, 3);

	FillMatrixWithRandomNumbers(arr2, 3, 3);

	cout << "\n Matrix Two: \n";
	PrintMatrix(arr2, 3, 3);

	MultiplyMatrix(arr, arr2, arrMultiply, 3, 3);

	cout << "\n Results\n";

	PrintMatrix(arrMultiply, 3, 3);


	return 0;
}
