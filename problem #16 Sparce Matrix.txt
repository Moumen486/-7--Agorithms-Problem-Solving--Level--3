{

}

int CountNumberInMatrix(int arr[3][3],int Number ,short Rows, short Cols)
{
	int counter = 0;
		for (short  i = 0; i < Rows; i++)
		{
			for (short j = 0; j < Cols; j++)
			{
				if (arr[i][j] == Number)
				{
					counter++;
				}
			}
		}
		return counter;
}
bool IsMatrixSparcy(int arr[3][3], short Rows, short Cols)
{
	short MatrixSize = Cols * Rows;
	return(CountNumberInMatrix(arr, 0,3, 3) >= (MatrixSize/2));
}
