void PrintFirstLettreOfEachWord(string S1)
{
	bool IsFirstLettre = true;
	cout << "\n First Letters Of This String: \n";

	{
		if (S1[i] != ' ' && IsFirstLettre)
		{
			cout << S1[i] << endl;
	   }

	}
}
