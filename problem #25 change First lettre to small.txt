string LowerCaseLetters(string S1)
{
	bool IsFirstLettre = true;
	for (short i = 0; i < S1.length(); i++)
	{

		if (S1[i] != ' ' && IsFirstLettre)
		{
			S1[i] = tolower(S1[i]);
	   }
		IsFirstLettre = (S1[i] == ' ' ? true : false);
	}
	return S1;
}
