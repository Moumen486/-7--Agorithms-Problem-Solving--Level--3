#include<iostream>
#include<string>
#include<iomanip>
using namespace std;
string ReadString()
{
	string S1;
	cout << "\n Please entre your string: \n";
	getline(cin, S1);
	
	return S1;
}
string UpperAllLetters(string S1)
{
	for (short i = 0; i < S1.length(); i++)
	{

		if (S1[i] != ' ')
		{
			S1[i] = toupper(S1[i]);
		}
	}
	return S1;
}
string LowerCaseAllLetters(string S1)
{
	for (short i = 0; i < S1.length(); i++)
	{

		if (S1[i] != ' ' )
		{
			S1[i] = tolower(S1[i]);
	   }
	}
	return S1;
}
int main()
{
	string S1 = ReadString();

	cout << "\n string after Upper: \n";
	cout << UpperAllLetters(S1) << endl;


	cout << "\n string after Lower: \n";
	cout << LowerCaseAllLetters(S1) << endl;
	

	return 0;
}
