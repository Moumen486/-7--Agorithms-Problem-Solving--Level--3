char InvertLettreCase(char S1)
{
	return isupper(S1) ? tolower(S1) : toupper(S1);

}
string InvertAllLetters(string S1)
{
	for (short i = 0; i < S1.length(); i++)
	{
		S1[i] = InvertLettreCase(S1[i]);
	}
	return S1;
}
