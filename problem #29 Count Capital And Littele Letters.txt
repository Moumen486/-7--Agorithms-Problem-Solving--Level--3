enum enWhatCount{SmallLetters=0,CapitalLetters=1};

short CountLetters(string S1, enWhatCount WhatCount)
{
	int count = 0;
	for (short i = 0; i < S1.length(); i++)
	{
		if (WhatCount == enWhatCount::CapitalLetters && isupper(S1[i]))
			count++;
		if (WhatCount == enWhatCount::SmallLetters && islower(S1[i]))
			count++;
	}
	return count;
}
int CountCapitaLletters(string S1)
{
	int count = 0;
	for (short i = 0; i <= S1.length(); i++)

int CountSmallletters(string S1)
{
	int count = 0;
	for (short i = 0; i <= S1.length(); i++)
	{
		if (islower(S1[i]))
		{
			count++;
		}
	}
	return count;
}
