char InvertLettreCase(char S2)
{
	return isupper(S2) ? tolower(S2) : toupper(S2);

}
int CountSmallCharachter(string S1,char S2,bool MatchCase=true )
{
	int count = 0;
	for (short i = 0; i < S1.length(); i++)
	{
		if (MatchCase)
		{
			if (S1[i] == S2)
				count++;
		}
		else
		{
			if (tolower(S1[i]) == tolower(S2))
				count++; 
		}
	}
	return count;
}
