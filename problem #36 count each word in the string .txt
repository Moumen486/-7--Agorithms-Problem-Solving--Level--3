void PrintWords(string S1)
{
	
	for (short i = 0; i < S1.length(); i++)
	{
		if (S1[i] != ' ')

			cout << S1[i];
		else
			cout << "\n";
	}
	
}
int CountWords(string S1)
{
	int counter = 0;
	string delim = " "; // delimiter
	short pos = 0;
	string sword;//define a string varaible 

	// use find() function to get the position of the delimiters
	while ((pos = S1.find(delim)) != std::string::npos)
	{
		sword = S1.substr(0, pos);//store the word
		if (sword != "")
		{
			counter++;
		}
		S1.erase(0, pos + delim.length()); //erase() until position and move the next word 
	}
	if (S1 != "")
	{
		counter++;// it print last word of th string	
	}

	return counter;
}
