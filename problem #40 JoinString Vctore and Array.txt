string JoinString(vector<string> vstring, string Delim)
{
	string S1="";
	for (string& s : vstring)
	{
		S1 = S1 + s + Delim;
	}

	return S1.substr(0, S1.length() - Delim.length());
}
string JoinStringarr(string arr, int length, string Delim)
{
	string S1 = "";
	for (short i = 0; i < length; i++)
	{
		S1 = S1 + arr[i] + Delim;
	}
	return S1.substr(0, S1.length() - Delim.length());
}
