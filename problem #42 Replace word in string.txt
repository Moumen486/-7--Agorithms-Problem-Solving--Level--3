string ReplaceWordInString(string S1,string StringToReplace,string ReplaceTo)
{
	short pos = S1.find(StringToReplace);
		while (pos != std::string::npos)
		{
			S1 = S1.replace(pos, StringToReplace.length(), ReplaceTo);
			pos = S1.find(StringToReplace);//find next

		}
		return S1;	
}
int main()
{	
	string S1 = "Welcome To Jordan,Jordan is a nice country";
	string StringToReplace = "jordan";
	string ReplaceTo = "USA";

	cout << "\n Original String :\n" << S1<<endl;
	cout << "\n String After Replacement:\n";
	cout << ReplaceWordInString(S1, StringToReplace, ReplaceTo) << endl;


	return 0;
}
